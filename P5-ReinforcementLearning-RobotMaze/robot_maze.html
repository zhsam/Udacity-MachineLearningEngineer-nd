def multi_runner(epoch, epsilon0, alpha, gamma, maze_size, trap_number):
    g = Maze(maze_size=maze_size,trap_number=trap_number)
    r = Robot(g,alpha=alpha, epsilon0=epsilon0, gamma=gamma)
    r.set_status(learning=True)

    runner = Runner(r, g)
    runner.run_training(epoch, display_direction=True)

    filename = 'final_'+str(datetime.today().month)+str(datetime.today().day)+'_'+str(datetime.today().second)+'.mp4'
    runner.generate_movie(filename) 

    runner.plot_results()
